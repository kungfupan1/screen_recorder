一、系统概述

1.1 设计目标
本系统旨在构建一个多应用共享的授权管理平台，支持：
多个App共用同一套授权管理后台
价格、方案可通过管理后台动态调整（无需修改代码）
支付与激活全自动完成
支持离线注册码激活（无需在线支付）

1.2 核心设计原则
原则             说明
反硬编码 所有价格、有效期、配置参数从数据库或环境变量读取
解耦设计 公开API与管理后台API分离，可独立部署
多App支持 通过 app_code 区分不同应用
离线激活 支持注册码激活，无需在线支付
容灾设计 客户端支持本地缓存和离线验签

-- ========== 应用表 ==========
CREATE TABLE apps (
    id INT PRIMARY KEY AUTO_INCREMENT,
    app_code VARCHAR(32) UNIQUE NOT NULL COMMENT '应用代码',
    app_name VARCHAR(64) NOT NULL COMMENT '应用名称',
    description VARCHAR(256),
    payment_channel_id INT DEFAULT NULL COMMENT '默认支付渠道ID',
    is_active BOOLEAN DEFAULT TRUE,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) COMMENT '应用表';

-- ========== 支付渠道表 ==========
CREATE TABLE payment_channels (
    id INT PRIMARY KEY AUTO_INCREMENT,
    channel_code VARCHAR(32) UNIQUE NOT NULL COMMENT '渠道代码(wechat/alipay等)',
    channel_name VARCHAR(64) NOT NULL COMMENT '渠道名称',
    channel_type VARCHAR(32) NOT NULL COMMENT '渠道类型(wechat_pay/alipay等)',
    config JSON NOT NULL COMMENT '渠道配置(密钥等，按渠道类型不同)',
    is_active BOOLEAN DEFAULT TRUE,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) COMMENT '支付渠道表';

-- config字段说明（JSON）:
-- 微信支付: {"mch_id": "", "api_key": "", "notify_url": ""}
-- 支付宝:   {"app_id": "", "private_key": "", "notify_url": ""}

-- ========== 授权方案表 ==========
CREATE TABLE license_plans (
    id INT PRIMARY KEY AUTO_INCREMENT,
    app_id INT NOT NULL,
    plan_code VARCHAR(32) NOT NULL COMMENT '方案代码',
    plan_name VARCHAR(64) NOT NULL COMMENT '方案名称',
    price DECIMAL(10, 2) NOT NULL COMMENT '价格（元）',
    duration_days INT NOT NULL COMMENT '有效期天数',
    sort_order INT DEFAULT 0,
    is_active BOOLEAN DEFAULT TRUE,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UNIQUE KEY unique_app_plan (app_id, plan_code)
) COMMENT '授权方案表';

-- ========== 订单表 ==========
CREATE TABLE orders (
    id INT PRIMARY KEY AUTO_INCREMENT,
    order_no VARCHAR(64) UNIQUE NOT NULL,
    app_id INT NOT NULL,
    plan_id INT NOT NULL,
    cpu_id VARCHAR(64) COMMENT '用户机器码',
    amount DECIMAL(10, 2) NOT NULL,
    status VARCHAR(20) DEFAULT 'pending' COMMENT 'pending/paid/closed',
    payment_channel VARCHAR(32) COMMENT '支付渠道代码(wechat/alipay等)',
    transaction_id VARCHAR(64) COMMENT '第三方交易号',
    license_code VARCHAR(256) COMMENT '生成的注册码',
    expire_date DATE,
    source VARCHAR(20) DEFAULT 'payment' COMMENT 'payment/keygen',
    paid_at DATETIME,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_order_no (order_no)
) COMMENT '订单表';

-- ========== 激活码表 ==========
CREATE TABLE licenses (
    id INT PRIMARY KEY AUTO_INCREMENT,
    app_id INT NOT NULL,
    license_code VARCHAR(256) UNIQUE NOT NULL COMMENT '注册码',
    cpu_id VARCHAR(64) COMMENT '绑定的机器码',
    plan_id INT NOT NULL,
    order_id INT,
    expire_date DATE NOT NULL,
    source VARCHAR(20) DEFAULT 'payment',
    is_revoked BOOLEAN DEFAULT FALSE,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_license_code (license_code),
    INDEX idx_cpu_id (cpu_id)
) COMMENT '激活码表';

-- ========== 管理员表 ==========
CREATE TABLE admins (
    id INT PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(32) UNIQUE NOT NULL,
    password_hash VARCHAR(128) NOT NULL,
    role VARCHAR(20) DEFAULT 'admin',
    last_login_at DATETIME,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
) COMMENT '管理员表';

-- ========== 系统配置表 ==========
CREATE TABLE system_config (
    id INT PRIMARY KEY AUTO_INCREMENT,
    config_key VARCHAR(64) UNIQUE NOT NULL,
    config_value TEXT,
    description VARCHAR(256),
    is_secret BOOLEAN DEFAULT FALSE,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) COMMENT '系统配置表';

五、注册码签名机制（Ed25519）★5.1 核心概念：签名与验签┌─────────────────────────────────────────────────────────────────────────────┐
│                     Ed25519 签名机制（非加密）                                │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                             │
│  【服务端】                           【客户端】                              │
│                                                                             │
│  私钥 (Private Key)                   公钥 (Public Key)                      │
│  ├─ 32字节                            ├─ 32字节                              │
│  ├─ 仅服务端保留                      ├─ 硬编码到客户端代码                   │
│  └─ 用于签名                          └─ 用于验签                            │
│                                                                             │
│  ────────────────────────────────────────────────────────────────────────  │
│                                                                             │
│  签名流程 (Sign):                                                          │
│  ┌─────────────────────────────────────────────────────────────────────┐   │
│  │  待签名数据 = 版本 + 方案ID + CPU哈希 + 到期时间戳                    │   │
│  │                   ↓                                                  │   │
│  │            Ed25519.Sign(私钥, 待签名数据)                            │   │
│  │                   ↓                                                  │   │
│  │              签名 (64字节)                                           │   │
│  │                   ↓                                                  │   │
│  │       注册码 = 待签名数据 + 签名                                      │   │
│  └─────────────────────────────────────────────────────────────────────┘   │
│                                                                             │
│  验签流程 (Verify):                                                        │
│  ┌─────────────────────────────────────────────────────────────────────┐   │
│  │  注册码 = Base64解码                                                 │   │
│  │       ↓                                                              │   │
│  │  提取: 待签名数据 + 签名                                              │   │
│  │       ↓                                                              │   │
│  │  Ed25519.Verify(公钥, 待签名数据, 签名)                              │   │
│  │       ↓                                                              │   │
│  │  真/假                                                               │   │
│  └─────────────────────────────────────────────────────────────────────┘   │
│                                                                             │
│  安全保证:                                                                   │
│  • 攻击者拥有公钥无法伪造签名                                                │
│  • 只有拥有私钥的服务端才能生成有效注册码                                     │
│  • 客户端可完全离线验签，无需请求服务器                                       │
│                                                                             │
└─────────────────────────────────────────────────────────────────────────────┘
5.2 注册码数据结构注册码二进制结构（78字节）

┌──────────┬──────────┬────────────────┬────────────┬──────────────┐
│ 偏移     │ 长度     │ 字段           │ 说明       │ 示例值       │
├──────────┼──────────┼────────────────┼────────────┼──────────────┤
│ 0        │ 1字节    │ version        │ 版本号     │ 0x03 (v3)    │
│ 1        │ 1字节    │ plan_id        │ 方案ID     │ 0x02         │
│ 2        │ 8字节    │ cpu_hash       │ CPU哈希    │ ABCD1234...  │
│ 10       │ 4字节    │ expire_ts      │ 到期时间戳  │ 1714848000   │
│ 14       │ 64字节   │ signature      │ Ed25519签名 │ (二进制)     │
└──────────┴──────────┴────────────────┴────────────┴──────────────┘

总长度: 1 + 1 + 8 + 4 + 64 = 78字节

Base64编码后: 约104字符

最终格式: REC-{Base64字符串}
示例: REC-YWJjZDEyMzRlZmdoNTY3OGlqa2w5MDEybW5vcDM0NTZxcnN0Nzg5MHV2d3gxMjM0eXphYjU2NzhjZGVmOTAxMmdoa2wzNDU2bW5vcDc4OTA
5.3 注册码签名模块（服务端）# ============================================================
# 文件: shared/crypto/license_signer.py
# 说明: Ed25519签名模块 - 仅服务端使用
# ============================================================

"""
Ed25519 签名模块（服务端专用）

核心概念:
- 这是签名(Sign)操作，不是加密(Encrypt)
- 私钥用于签名，公钥用于验签
- 私钥绝对不能泄露给客户端
"""

import base64
import struct
import time
from datetime import datetime, timedelta
from typing import Tuple

from cryptography.hazmat.primitives.asymmetric.ed25519 import (
    Ed25519PrivateKey,
    Ed25519PublicKey
)
from cryptography.hazmat.primitives import serialization


class LicenseSigner:
    """注册码签名器 - 服务端使用"""

    # 数据结构常量
    VERSION = 3
    DATA_LENGTH = 14      # 版本(1) + 方案(1) + CPU哈希(8) + 时间戳(4)
    SIGNATURE_LENGTH = 64  # Ed25519签名固定64字节
    TOTAL_LENGTH = 78      # DATA_LENGTH + SIGNATURE_LENGTH

    @staticmethod
    def generate_key_pair() -> Tuple[str, str]:
        """
        生成Ed25519密钥对

        返回: (private_key_base64, public_key_base64)

        使用场景: 系统初始化时执行一次，私钥存数据库，公钥给客户端
        """
        private_key = Ed25519PrivateKey.generate()
        public_key = private_key.public_key()

        # 私钥编码
        private_bytes = private_key.private_bytes(
            encoding=serialization.Encoding.Raw,
            format=serialization.PrivateFormat.Raw,
            encryption_algorithm=serialization.NoEncryption()
        )

        # 公钥编码
        public_bytes = public_key.public_bytes(
            encoding=serialization.Encoding.Raw,
            format=serialization.PublicFormat.Raw
        )

        return (
            base64.b64encode(private_bytes).decode('utf-8'),
            base64.b64encode(public_bytes).decode('utf-8')
        )

    @staticmethod
    def sign(
        machine_code: str,
        plan_id: int,
        duration_days: int,
        private_key_b64: str
    ) -> str:
        """
        生成注册码（对数据进行签名）

        参数:
            machine_code: 用户机器码（32字符十六进制，如 "ABCD1234EFGH5678"）
            plan_id: 方案ID（1-255）
            duration_days: 有效期天数
            private_key_b64: Base64编码的Ed25519私钥

        返回:
            格式化的注册码字符串，以 "REC-" 开头

        签名流程:
            1. 组装待签名数据（14字节）
            2. 使用Ed25519私钥签名（64字节）
            3. 合成完整数据（78字节）
            4. Base64编码，加前缀
        """
        # 1. 解析私钥
        private_bytes = base64.b64decode(private_key_b64)
        private_key = Ed25519PrivateKey.from_private_bytes(private_bytes)

        # 2. 处理机器码：取前16字符，转为8字节
        clean_code = machine_code.replace('-', '').upper()
        if len(clean_code) < 16:
            clean_code = clean_code.ljust(16, '0')
        cpu_hash_bytes = bytes.fromhex(clean_code[:16])

        # 3. 计算到期时间戳
        expire_date = datetime.now() + timedelta(days=duration_days)
        expire_ts = int(expire_date.timestamp())

        # 4. 组装待签名数据（14字节）
        # 结构: 版本(1) + 方案ID(1) + CPU哈希(8) + 时间戳(4)
        data_to_sign = bytes([
            LicenseSigner.VERSION,      # 版本号
            plan_id                     # 方案ID
        ]) + cpu_hash_bytes + struct.pack('>I', expire_ts)  # 大端序时间戳

        # 5. Ed25519签名
        signature = private_key.sign(data_to_sign)

        # 6. 合成完整数据
        full_data = data_to_sign + signature

        # 7. Base64编码，去除特殊字符
        encoded = base64.b64encode(full_data).decode('utf-8')
        # 替换URL不安全字符
        encoded = encoded.replace('+', '-').replace('/', '_')

        # 8. 添加前缀并返回
        return f"REC-{encoded}"

    @staticmethod
    def verify_signature(
        license_code: str,
        public_key_b64: str
    ) -> dict:
        """
        验证注册码签名（服务端可用，但客户端应使用LicenseVerifier）

        此方法主要用于服务端调试，客户端应使用 license_verifier.py
        """
        from shared.crypto.license_verifier import LicenseVerifier
        return LicenseVerifier.verify(license_code, public_key_b64)
5.4 注册码验签模块（客户端）# ============================================================
# 文件: shared/crypto/license_verifier.py
# 说明: Ed25519验签模块 - 客户端使用
# ============================================================

"""
Ed25519 验签模块（客户端专用）

核心概念:
- 这是验签(Verify)操作，不是解密(Decrypt)
- 只需要公钥即可验签
- 攻击者拥有公钥无法伪造签名
"""

import base64
import struct
from datetime import datetime
from typing import Optional

from cryptography.hazmat.primitives.asymmetric.ed25519 import Ed25519PublicKey
from cryptography.exceptions import InvalidSignature


class LicenseVerifier:
    """注册码验签器 - 客户端使用"""

    # 数据结构常量（与签名器一致）
    DATA_LENGTH = 14
    SIGNATURE_LENGTH = 64
    TOTAL_LENGTH = 78
    PREFIX = "REC-"

    @staticmethod
    def verify(
        license_code: str,
        public_key_b64: str
    ) -> dict:
        """
        验证注册码（纯离线验签）

        参数:
            license_code: 注册码字符串（以REC-开头）
            public_key_b64: Base64编码的Ed25519公钥（硬编码在客户端）

        返回:
            {
                "valid": bool,
                "plan_id": int,
                "expire_date": str,
                "expire_ts": int,
                "days_left": int,
                "message": str
            }
        """
        try:
            # 1. 检查前缀
            if not license_code.startswith(LicenseVerifier.PREFIX):
                return {"valid": False, "message": "注册码格式错误：缺少前缀"}

            # 2. Base64解码
            encoded = license_code[4:]  # 去掉 "REC-"
            # 还原URL安全字符
            encoded = encoded.replace('-', '+').replace('_', '/')
            # 补齐填充
            padding = 4 - len(encoded) % 4
            if padding != 4:
                encoded += '=' * padding

            full_data = base64.b64decode(encoded)

            # 3. 检查长度
            if len(full_data) != LicenseVerifier.TOTAL_LENGTH:
                return {"valid": False, "message": f"注册码长度错误：{len(full_data)}"}

            # 4. 提取字段
            version = full_data[0]
            plan_id = full_data[1]
            cpu_hash = full_data[2:10]
            expire_ts = struct.unpack('>I', full_data[10:14])[0]
            signature = full_data[14:78]

            # 5. 检查版本
            if version != 3:
                return {"valid": False, "message": f"注册码版本不支持：v{version}"}

            # 6. 验签
            public_bytes = base64.b64decode(public_key_b64)
            public_key = Ed25519PublicKey.from_public_bytes(public_bytes)

            data_signed = full_data[:14]

            try:
                public_key.verify(signature, data_signed)
            except InvalidSignature:
                return {"valid": False, "message": "注册码签名无效"}

            # 7. 检查到期时间
            expire_date = datetime.fromtimestamp(expire_ts)
            now = datetime.now()
            days_left = (expire_date - now).days

            if days_left <= 0:
                return {"valid": False, "message": "注册码已过期", "expire_date": expire_date.strftime("%Y-%m-%d")}

            # 8. 返回成功结果
            return {
                "valid": True,
                "plan_id": plan_id,
                "cpu_hash": cpu_hash.hex().upper(),
                "expire_date": expire_date.strftime("%Y-%m-%d"),
                "expire_ts": expire_ts,
                "days_left": days_left,
                "message": "验签成功"
            }

        except Exception as e:
            return {"valid": False, "message": f"注册码解析错误：{str(e)}"}

    @staticmethod
    def quick_verify(license_code: str, public_key_b64: str) -> bool:
        """
        快速验签（只返回是否有效）

        用于启动时快速检查本地缓存
        """
        result = LicenseVerifier.verify(license_code, public_key_b64)
        return result.get("valid", False)
5.5 机器码生成（客户端）# ============================================================
# 文件: shared/crypto/machine_code.py
# 说明: 机器码生成模块
# ============================================================

"""
机器码生成模块

机器码 = SHA256(CPU序列号) 的前16字节 = 32个十六进制字符
显示格式: ABCD-1234-EFGH-5678（4组，每组4字符）
"""

import hashlib
import subprocess
import platform


def get_cpu_id() -> str:
    """
    获取CPU序列号（Windows）

    返回: 原始CPU序列号
    """
    if platform.system() != 'Windows':
        raise NotImplementedError("仅支持Windows系统")

    try:
        result = subprocess.run(
            ['wmic', 'cpu', 'get', 'ProcessorId'],
            capture_output=True,
            text=True,
            creationflags=subprocess.CREATE_NO_WINDOW
        )

        lines = result.stdout.strip().split('\n')
        if len(lines) >= 2:
            return lines[1].strip()
    except Exception:
        pass

    # 备用方案
    return "UNKNOWN_CPU"


def get_machine_code() -> str:
    """
    获取机器码

    返回: 32字符十六进制字符串，格式化显示
    """
    cpu_id = get_cpu_id()

    # SHA256哈希
    hash_value = hashlib.sha256(cpu_id.encode()).hexdigest()

    # 取前32字符
    machine_code = hash_value[:32].upper()

    return machine_code


def get_machine_code_display() -> str:
    """
    获取格式化的机器码（用于UI显示）

    返回: ABCD-1234-EFGH-5678 格式
    """
    code = get_machine_code()
    return '-'.join([code[i:i+4] for i in range(0, 16, 4)])


def parse_machine_code(display_code: str) -> str:
    """
    解析格式化的机器码

    参数: ABCD-1234-EFGH-5678 格式
    返回: ABCD1234EFGH5678（16字符）
    """
    return display_code.replace('-', '').upper()[:16]

# 文件: 客户端 license/cache_manager.py

"""
本地缓存管理器

import os
import json
from datetime import datetime
from typing import Optional, Dict

# 缓存目录
CACHE_DIR = os.path.join(
    os.path.expanduser("~"),
    "AppData",
    "Local",
    "录屏王"
)
LICENSE_FILE = os.path.join(CACHE_DIR, "license.json")
PLANS_CACHE_FILE = os.path.join(CACHE_DIR, "plans_cache.json")


def ensure_cache_dir():
    """确保缓存目录存在"""
    if not os.path.exists(CACHE_DIR):
        os.makedirs(CACHE_DIR)


def save_license(
    license_code: str,
    machine_code: str,
    plan_name: str = "",
    expire_date: str = ""
):
    """
    保存激活信息到本地（强制覆写）

    重要: 此函数总是覆盖现有文件，不做任何合并或询问

    参数:
        license_code: 注册码（REC-开头）
        machine_code: 当前机器码
        plan_name: 方案名称
        expire_date: 到期日期

    ensure_cache_dir()

    data = {
        "license_code": license_code,
        "machine_code": machine_code,
        "plan_name": plan_name,
        "expire_date": expire_date,
        "activated_at": datetime.now().strftime("%Y-%m-%dT%H:%M:%S")
    }

    # 强制覆写
    with open(LICENSE_FILE, 'w', encoding='utf-8') as f:
        json.dump(data, f, ensure_ascii=False, indent=2)

    加载本地激活信息

    返回: 激活信息字典，不存在则返回None

            return json.load(f)

        return None

    """清除本地激活信息"""
    if os.path.exists(LICENSE_FILE):
        os.remove(LICENSE_FILE)

        return None

def check_activation() -> Dict:
    """
    检查激活状态（启动时调用）

    返回:
        {
            "activated": bool,
            "plan_name": str,
            "expire_date": str,
            "days_left": int,
            "message": str
        }

    license_data = load_license()

        "expire_date": result.get("expire_date"),

        "success": True,
        "message": "激活成功",
        "expire_date": result.get("expire_date"),
        "days_left": result.get("days_left")
    }

"""
注册码生成器API

核心功能:
- 使用Ed25519私钥对数据进行签名
- 生成注册码供用户离线激活
"""

from flask import Blueprint, request, jsonify
from shared.crypto.license_signer import LicenseSigner
from shared.crypto.machine_code import parse_machine_code
from shared.database.models import get_db_connection
from utils.auth import login_required

keygen_bp = Blueprint('keygen', __name__, url_prefix='/admin/keygen')


@keygen_bp.route('/generate', methods=['POST'])
@login_required
def generate():
    """
    生成注册码

    请求:
        {
            "app_code": "screen_recorder",
            "machine_code": "ABCD-1234-EFGH-5678",
            "plan_id": 2
        }

    响应:
        {
            "success": true,
            "data": {
                "license_code": "REC-...",
                "plan_name": "月度授权包",
                "expire_date": "2025-05-03",
                "machine_code": "ABCD-1234-EFGH-5678"
            }
        }
    """
    data = request.get_json()

    app_code = data.get('app_code')
    machine_code_display = data.get('machine_code', '')
    plan_id = data.get('plan_id')

    if not machine_code_display or not plan_id:
        return jsonify({"success": False, "error": "参数不完整"}), 400

    # 解析机器码
    machine_code = parse_machine_code(machine_code_display)

    # 从数据库获取方案信息和私钥
    conn = get_db_connection()

    # 获取方案
    cursor = conn.execute("""
        SELECT lp.plan_name, lp.duration_days, lp.app_id
        FROM license_plans lp
        JOIN apps a ON lp.app_id = a.id
        WHERE a.app_code = ? AND lp.id = ?
    """, (app_code, plan_id))
    plan = cursor.fetchone()

    if not plan:
        return jsonify({"success": False, "error": "方案不存在"}), 400

    plan_name, duration_days, app_id = plan

    # 获取私钥
    cursor = conn.execute("""
        SELECT config_value FROM system_config WHERE config_key = 'license_private_key'
    """)
    row = cursor.fetchone()
    private_key = row[0] if row else None

    if not private_key:
        return jsonify({"success": False, "error": "私钥未配置"}), 500

    # 签名生成注册码
    try:
        license_code = LicenseSigner.sign(
            machine_code=machine_code,
            plan_id=plan_id,
            duration_days=duration_days,
            private_key_b64=private_key
        )
    except Exception as e:
        return jsonify({"success": False, "error": f"签名失败: {str(e)}"}), 500

    # 计算到期日期
    from datetime import datetime, timedelta
    expire_date = (datetime.now() + timedelta(days=duration_days)).strftime("%Y-%m-%d")

    return jsonify({
        "success": True,
        "data": {
            "license_code": license_code,
            "plan_name": plan_name,
            "expire_date": expire_date,
            "machine_code": machine_code_display
        }
    })

示例:
